### RGB Colors (24-bit)

True color with 16 million colors:

```typescript
interface RgbColor {
  readonly type: 'rgb';
  readonly r: number;
  readonly g: number;
  readonly b: number;
}

const rgbColor = (r: number, g: number, b: number): Color => {
  if (r < 0 || r > 255 || g < 0 || g > 255 || b < 0 || b > 255) {
    throw new Error(`rgb values must be 0-255, got (${r}, ${g}, ${b})`);
  }
  return { type: 'rgb', r, g, b };
};
```

## Modifiers

Text modifiers are bitflags:

```typescript
const Modifier = {
  NONE: 0,
  BOLD: 1,
  DIM: 2,
  ITALIC: 4,
  UNDERLINED: 8,
  SLOW_BLINK: 16,
  RAPID_BLINK: 32,
  REVERSED: 64,
  HIDDEN: 128,
  CROSSED_OUT: 256,
  DOUBLE_UNDERLINED: 512,
  OVERLINED: 1024,
  ALL: /* bitwise OR of all flags */,
  contains: (flags: number, flag: number): boolean => (flags & flag) === flag,
  add: (flags: number, flag: number): number => flags | flag,
  remove: (flags: number, flag: number): number => flags & ~flag,
  union: (a: number, b: number): number => a | b,
} as const;
```

## Style Composition

The `patchStyle` function merges two styles:

```typescript
const patchStyle = (base: Style, patch: Style): Style => {
  const fg = patch.fg ?? base.fg;
  const bg = patch.bg ?? base.bg;
  const underlineColor = patch.underlineColor ?? base.underlineColor;
  const addModifier = (base.addModifier | patch.addModifier) & ~patch.subModifier;
  const subModifier = (base.subModifier | patch.subModifier) & ~patch.addModifier;

  return {
    ...(fg !== undefined ? { fg } : {}),
    ...(bg !== undefined ? { bg } : {}),
    ...(underlineColor !== undefined ? { underlineColor } : {}),
    addModifier,
    subModifier,
  };
};
```

From `~/workspace/source/src/core/style.ts:163-177`

**Example:**
```typescript
import { createStyle, styleFg, styleAddModifier, patchStyle, Color, Modifier } from 'terminui';

const bold = styleAddModifier(createStyle(), Modifier.BOLD);
const redText = styleFg(createStyle(), Color.Red);
const boldRed = patchStyle(bold, redText);
```

<Tip>
`patchStyle` is used internally when rendering styled text. The patch overwrites the base for colors, and merges modifiers.
</Tip>

## Complete Example

Building a complex style:

```typescript
import {
  createStyle,
  styleFg,
  styleBg,
  styleAddModifier,
  patchStyle,
  rgbColor,
  Color,
  Modifier,
} from 'terminui';

// Base style: bold
const bold = styleAddModifier(createStyle(), Modifier.BOLD);

// Warning style: yellow on black
const warning = styleBg(styleFg(createStyle(), Color.Yellow), Color.Black);

// Merge: bold yellow on black
const boldWarning = patchStyle(bold, warning);

// Custom RGB color
const teal = rgbColor(0, 128, 128);
const tealBold = patchStyle(bold, styleFg(createStyle(), teal));

// Multiple modifiers
const fancy = styleAddModifier(
  styleAddModifier(createStyle(), Modifier.BOLD),
  Modifier.ITALIC,
);
```

## Using Styles with Cells

Styles are applied to cells using `cellSetStyle`:

```typescript
const cellSetStyle = (cell: Cell, style: Style): Cell => {
  const fg = style.fg ?? cell.fg;
  const bg = style.bg ?? cell.bg;
  const underlineColor = style.underlineColor ?? cell.underlineColor;
  const modifier = (cell.modifier | style.addModifier) & ~style.subModifier;

  return {
    symbol: cell.symbol,
    fg,
    bg,
    ...(underlineColor !== undefined ? { underlineColor } : {}),
    modifier,
  };
};
```
